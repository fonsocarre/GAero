265 void GAclass::crossIndividuals()
266 {
267  std::vector<double> weights;
268  weights = oldPopFitness2vec();
269 
270  for (int i=0; i<this->GAsettings.nCrossing; i++)
271  {
272  // newPop position of the created individual
273  int iNewPop = i + this->usedPopulation;
274  // roulette selection of the two parents
275  int index1 = this->randomGen.roulette (weights);
276  int index2 = this->randomGen.roulette (weights);
277  if (!(this->GAsettings.nCrossing == 1))
278  {
279  while (index1 == index2)
280  {
281  index2 = this->randomGen.roulette (weights);
282  }
283  }
284 
285  if ((index1==-1) || (index2==-1))
286  {this->GAout<<"ROULETTE ERROR"<<std::endl;}
287 
288  double fitness1 = this->oldPopulation[index1].fitness;
289  double fitness2 = this->oldPopulation[index2].fitness;
290  utilities::normalizeFitness(fitness1, fitness2);
291 
292  // genome filling
293  for (int j=0; j<this->GAsettings.genomeSize; j++)
294  {
295  this->population[iNewPop].genome[j] =
296  fitness1*this->oldPopulation[index1].genome[j] +
297  fitness2*this->oldPopulation[index2].genome[j];
298  }
299  // other attributes
300  this->population[iNewPop].generation = this->iGeneration;
301  this->population[iNewPop].genomeLength = this->GAsettings.genomeSize;
302  this->population[iNewPop].isFitnessCalculated = false;
303  }
304  this->usedPopulation += this->GAsettings.nCrossing;
305 }
