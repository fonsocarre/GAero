50

52  Init0(19650218);

75 uint32_t CRandomMersenne::BRandom() {
76  // Generate 32 random bits
77  uint32_t y;
78 
79  if (mti >= MERS_N) {
80  // Generate MERS_N words at one time
81  const uint32_t LOWER_MASK = (1LU << MERS_R) - 1; // Lower MERS_R bits
82  const uint32_t UPPER_MASK = 0xFFFFFFFF << MERS_R; // Upper (32 - MERS_R) bits
83  static const uint32_t mag01[2] = {0, MERS_A};
84 
85  int kk;
86  for (kk=0; kk < MERS_N-MERS_M; kk++) {
87  y = (mt[kk] & UPPER_MASK) | (mt[kk+1] & LOWER_MASK);
88  mt[kk] = mt[kk+MERS_M] ^ (y >> 1) ^ mag01[y & 1];}
89 
90  for (; kk < MERS_N-1; kk++) {
91  y = (mt[kk] & UPPER_MASK) | (mt[kk+1] & LOWER_MASK);
92  mt[kk] = mt[kk+(MERS_M-MERS_N)] ^ (y >> 1) ^ mag01[y & 1];}
93 
94  y = (mt[MERS_N-1] & UPPER_MASK) | (mt[0] & LOWER_MASK);
95  mt[MERS_N-1] = mt[MERS_M-1] ^ (y >> 1) ^ mag01[y & 1];
96  mti = 0;
97  }
98  y = mt[mti++];
99 
100  // Tempering (May be omitted):
101  y ^= y >> MERS_U;
102  y ^= (y << MERS_S) & MERS_B;
103  y ^= (y << MERS_T) & MERS_C;
104  y ^= y >> MERS_L;
105 
106  return y;
107 }

117 int CRandomMersenne::IRandom(int min, int max) {
118  // Output random integer in the interval min <= x <= max
119  // Relative error on frequencies < 2^-32
120  if (max <= min) {
121  if (max == min) return min; else return 0x80000000;
122  }

130 int CRandomMersenne::IRandomX(int min, int max) {
131  // Output random integer in the interval min <= x <= max
132  // Each output value has exactly the same probability.
133  // This is obtained by rejecting certain bit values so that the number
134  // of possible bit values is divisible by the interval length
135  if (max <= min) {
136  if (max == min) return min; else return 0x80000000;
137  }
138 #ifdef INT64_SUPPORTED
139  // 64 bit integers available. Use multiply and shift method
140  uint32_t interval; // Length of interval
141  uint64_t longran; // Random bits * interval
142  uint32_t iran; // Longran / 2^32
143  uint32_t remainder; // Longran % 2^32
144 
145  interval = uint32_t(max - min + 1);
146  if (interval != LastInterval) {
147  // Interval length has changed. Must calculate rejection limit
148  // Reject when remainder >= 2^32 / interval * interval
149  // RLimit will be 0 if interval is a power of 2. No rejection then
150  RLimit = uint32_t(((uint64_t)1 << 32) / interval) * interval - 1;
151  LastInterval = interval;
152  }
153  do { // Rejection loop
154  longran = (uint64_t)BRandom() * interval;
155  iran = (uint32_t)(longran >> 32);
156  remainder = (uint32_t)longran;
157  } while (remainder > RLimit);
158  // Convert back to signed and return result
159  return (int32_t)iran + min;
160 
161 #else
162  // 64 bit integers not available. Use modulo method
163  uint32_t interval; // Length of interval
164  uint32_t bran; // Random bits
165  uint32_t iran; // bran / interval
166  uint32_t remainder; // bran % interval
167 
168  interval = uint32_t(max - min + 1);
169  if (interval != LastInterval) {
170  // Interval length has changed. Must calculate rejection limit
171  // Reject when iran = 2^32 / interval
172  // We can't make 2^32 so we use 2^32-1 and correct afterwards
173  RLimit = (uint32_t)0xFFFFFFFF / interval;
174  if ((uint32_t)0xFFFFFFFF % interval == interval - 1) RLimit++;
175  }
176  do { // Rejection loop
177  bran = BRandom();
178  iran = bran / interval;
179  remainder = bran % interval;
180  } while (iran >= RLimit);
181  // Convert back to signed and return result
182  return (int32_t)remainder + min;
183 
184 #endif
185 }
