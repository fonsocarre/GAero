621 inline void
622 Configuration::parse(const char * str)
623  throw(ConfigurationException)
624 {
625  if (strncmp(str, "exec#", 5) == 0) {
626  parse(Configuration::INPUT_EXEC, &(str[5]));
627  } else if (strncmp(str, "file#", 5) == 0) {
628  parse(Configuration::INPUT_FILE, &(str[5]));
629  } else {
630  parse(Configuration::INPUT_FILE, str);
631  }
